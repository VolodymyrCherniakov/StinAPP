105global_stock_cache = {} 

106cache_lock = Lock() 

107 

108def declined_last_3_days(prices): 

109 return bool(all(prices[i]['close'] < prices[i + 1]['close'] for i in range(3))) 

110 

111def more_than_two_declines_in_last_5_days(prices): 

112 return bool(sum(prices[i]['close'] < prices[i + 1]['close'] for i in range(5)) > 2) 

113 

114def get_stock_entry(ticker, use_cache_only=False, force_refresh=False): 

115 with cache_lock: 

116 if ticker in global_stock_cache and not force_refresh: 

117 return global_stock_cache[ticker] 

118 elif use_cache_only: 

119 return {"error": f"Data pro {ticker} nejsou v cache."} 

120 

121 try: 

122 end_date = datetime.today() 

123 start_date = end_date - timedelta(days=10) 

124 historical_prices = client.get_dataframe(ticker, startDate=start_date.strftime('%Y-%m-%d'), endDate=end_date.strftime('%Y-%m-%d')) 

125 

126 if historical_prices.empty or len(historical_prices) < 5: 

127 return {"error": f"Nedostatek dat pro {ticker}"} 

128 

129 latest_data = historical_prices.sort_index(ascending=False).head(6) 

130 prices = [{"date": date.strftime('%Y-%m-%d'), "close": row['close']} for date, row in latest_data.iterrows()] 

131 

132 result_entry = { 

133 "company_name": TICKER_NAMES.get(ticker, 'Neznámá společnost'), 

134 "declined_last_3_days": declined_last_3_days(prices), 

135 "more_than_2_declines_last_5_days": more_than_two_declines_in_last_5_days(prices), 

136 "latest_close": float(prices[0]['close']), 

137 "history": prices 

138 } 

139 

140 with cache_lock: 

141 global_stock_cache[ticker] = result_entry 

142 

143 return result_entry 

144 

145 except requests.exceptions.HTTPError as http_err: 

146 if http_err.response.status_code == 404: 

147 return {"error": f"Ticker '{ticker}' nebyl nalezen."} 

148 return {"error": f"HTTP chyba: {str(http_err)}"} 

149 except Exception as e: 

150 logger.error(f"Chyba při načítání {ticker}: {e}") 

151 return {"error": "Nastala chyba při získávání dat."}

153def scheduled_stock_fetch(): 

154 logger.info("Spouštím aktualizaci tickerů v cache...") 

155 with cache_lock: 

156 tickers_to_update = list(global_stock_cache.keys()) 

157 

158 for ticker in tickers_to_update: 

159 try: 

160 updated_entry = get_stock_entry(ticker, force_refresh=True) 

161 with cache_lock: 

162 global_stock_cache[ticker] = updated_entry 

163 logger.info(f"[REFRESH] Načítám znovu data pro {ticker}") 

164 except Exception as e: 

165 logger.warning(f"Chyba při aktualizaci {ticker}: {e}")

228@app.route('/api/news', methods=['POST']) 

229def fetch_and_filter_news(): 

230 try: 

231 body = request.get_json() 

232 api_url = body.get('api_url') 

233 min_rating = body.get('min_rating_for_sell', 0) 

234 

235 if not api_url: 

236 return jsonify({'error': 'Chybí api_url'}), 400 

237 

238 # Načtení zpráv z externího API 

239 response = requests.get(api_url) 

240 logger.info(f"[NEWS] {response}") 

241 if response.status_code != 200: 

242 return jsonify({'error': 'Nepodařilo se stáhnout data ze zadané adresy'}), 500 

243 

244 all_news = response.json() 

245 

246 # Filtrování a označování 

247 filtered = [] 

248 for item in all_news: 

249 rating = item.get('rating') 

250 if rating is None: 

251 continue 

252 

253 new_item = { 

254 'name': item.get('name'), 

255 'date': item.get('date'), 

256 'rating': rating, 

257 'sell': 1 if rating < min_rating else 0 

258 } 

259 filtered.append(new_item) 

260 logger.info(f"[NEWS] Načítám zprávy z: {api_url} | Min. hodnocení pro SELL: {min_rating}") 

261 logger.info(f"[NEWS] Načteno {len(all_news)} zpráv, filtrovaných: {len(filtered)}") 

262 return jsonify({'data': filtered}) 

263 

264 except Exception as e: 

265 print('Chyba:', e) 

266 return jsonify({'error': 'Došlo k chybě při zpracování'}), 500
